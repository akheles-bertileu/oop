class Robot:
    def __init__(self, model, power):
        self.model = model
        self.power = power

    def charge(self, amount):
        self.power = min(100, self.power + amount)
        print(f"{self.model} қуатталды! Қазір қуаты: {self.power}%")

    def work(self):
        if self.power <= 10:
            print(f"{self.model}: Қуат аз! Заряд керек ⚡️")
        else:
            self.power -= 10
            print(f"{self.model} жұмыс істеді. Қалған қуат: {self.power}%")

bot = Robot("Atlas", 50)
bot.work()
bot.charge(30)
bot.work()

Жауабы:
Atlas жұмыс істеді. Қалған қуат: 40%
Atlas қуатталды! Қазір қуаты: 70%
Atlas жұмыс істеді. Қалған қуат: 60%

class Robot:
    def __init__(self, model, power):
        self._model = model
        self._power = power  

    @property
    def power(self):
        return self._power

    @power.setter
    def power(self, value):
        if not (0 <= value <= 100):
            raise ValueError("Қуат деңгейі 0 мен 100 арасында болуы керек!")
        self._power = value

bot = Robot("Atlas", 50)
print(bot.power) 
bot.power = 80   
print(bot.power)  

Жауабы:
50
80

class Robot:
    def __init__(self, model, power):
        self._model = model
        self._power = power  

    @property
    def power(self):
        return self._power

    @power.setter
    def power(self, value):
        if not (0 <= value <= 100):
            raise ValueError("Қуат деңгейі 0 мен 100 арасында болуы керек!")
        self._power = value

class AdvancedRobot(Robot):
    def boost_power(self, percent):
        """Қуатты пайызбен арттыру"""
        self.power += self.power * percent
        if self.power > 100:
            self.power = 100

bot = AdvancedRobot("Atlas", 50)
bot.boost_power(0.2)  
print(bot.power)

Жауабы:
60.0
