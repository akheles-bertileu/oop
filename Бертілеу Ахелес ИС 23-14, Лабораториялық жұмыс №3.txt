def reverse_list(lst):
    return lst[::-1]

def list_sort(lst):
    return sorted(lst, key=abs, reverse=True)

def change(lst):
    lst[0], lst[-1] = lst[-1], lst[0]
    return lst

example_list = [3, -7, 0, 5, -2]

print("Исходный список:", example_list)
print("Обратный порядок:", reverse_list(example_list))
print("Сортировка по убыванию абсолютного значения:", list_sort(example_list))
print("После смены первого и последнего элемента:", change(example_list.copy()))
