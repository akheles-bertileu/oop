a = 10
b = 3
result_subtraction = a - b
print(f"Вычитание: {a} - {b} = {result_subtraction}")

result_addition = a + b
print(f"Сложение: {a} + {b} = {result_addition}")

result_multiplication = a * b
print(f"Умножение: {a} * {b} = {result_multiplication}")

result_exponentiation = a ** b
print(f"Возведение в степень: {a} ** {b} = {result_exponentiation}")

result_division = a / b
print(f"Деление: {a} / {b} = {result_division}")

result_integer_division = a // b
print(f"Целочисленное деление: {a} // {b} = {result_integer_division}")

result_modulo = a % b
print(f"Остаток от деления: {a} % {b} = {result_modulo}")
