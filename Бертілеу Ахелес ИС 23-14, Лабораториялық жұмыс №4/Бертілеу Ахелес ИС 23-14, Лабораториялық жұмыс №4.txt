def analyze_array(arr):
    negative_numbers = [x for x in arr if x < 0]
    positive_numbers = [x for x in arr if x > 0]
    
    if negative_numbers:
        avg_negative = sum(negative_numbers) / len(negative_numbers)
        min_negative = min(negative_numbers)
    else:
        avg_negative = None  
        min_negative = None
    
    if positive_numbers:
        min_positive = min(positive_numbers)
    else:
        min_positive = None  
    
    return negative_numbers, avg_negative, min_negative, min_positive

A = [3.5, -2.1, 0.0, 7.8, -5.6, 4.4, 1.0, -0.9, 2.2, -3.3]

def max_above_main_diagonal(matrix):
    n = len(matrix)
    elements = []
    
    for i in range(n):
        for j in range(i + 1, n):  
            elements.append(matrix[i][j])
    
    if elements:
        return max(elements)
    else:
        return None  


A = [
    [1,  5,  3,  2],
    [4,  6,  8,  7],
    [9,  0,  5,  1],
    [3,  4,  2,  9]
]

max_val = max_above_main_diagonal(A)
print(f"Максимум элементов над главной диагональю: {max_val}")
